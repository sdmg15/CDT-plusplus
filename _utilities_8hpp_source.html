313 {

315  std::random_device rd;
316  // The simple way which works in C++14
317  // std::mt19937_64 generator(rd());

322  // Copy into heap-allocated "seed sequence"

324  // Initialized mt19937_64

326 
327  Distribution distribution(min_value, max_value);
