234 /// @brief Seed sequence class for high-quality pseudo-random number generator
235 ///
236 /// From Arthur O'Dwyer's "Mastering the C++17 STL", Chapter 12
237 /// @tparam T1 Type of number
238 template <typename T1>
239 struct SeedSeq
240 {
241  T1 begin_;
242  T1 end_;
243 
244  public:
245  SeedSeq(T1 begin, T1 end) : begin_{begin}, end_{end} {}
246 
247  template <typename T2>
248  void generate(T2 b, T2 e)
249  {
250  assert((e - b) <= (end_ - begin_));
251  std::copy(begin_, begin_ + (e - b), b);
252  }
253 };

347 /// @brief Calculate expected # of points per simplex
348 ///
349 /// Usually, there are less vertices than simplices.
350 /// Here, we throw away a number of simplices that aren't correctly
351 /// foliated.
352 /// The exact formula is given by Dwyer:
353 /// http://link.springer.com/article/10.1007/BF02574694
354 ///
355 /// @param dimension Number of dimensions
356 /// @param simplices Number of desired simplices
357 /// @param timeslices Number of desired timeslices
358 /// @param output Prints desired number of simplices on timeslices
359 /// @return The number of points per timeslice to obtain
360 /// the desired number of simplices
361 inline auto expected_points_per_simplex(const int dimension,
362  const std::int32_t simplices,
363  const std::int32_t timeslices,
364  const bool output = true)
365 {
366  if (output)
367  {
368  std::cout << simplices << " simplices on " << timeslices
369  << " timeslices desired.\n";
370  }
371 
372  const auto simplices_per_timeslice = simplices / timeslices;
373  switch (dimension)
374  {
375  case 3:
376  {
377  // Avoid segfaults for small values
378  if (simplices == timeslices) { return 2 * simplices_per_timeslice; }
379  else if (simplices < 1000)
380  {
381  return static_cast<std::int32_t>(0.4 * simplices_per_timeslice);
382  }
383  else if (simplices < 10000)
384  {
385  return static_cast<std::int32_t>(0.2 * simplices_per_timeslice);
386  }
387  else if (simplices < 100000)
388  {
389  return static_cast<std::int32_t>(0.15 * simplices_per_timeslice);
390  }
391  else
392  {
393  return static_cast<std::int32_t>(0.1 * simplices_per_timeslice);
394  }
395  }
396  default:
397  {
398  throw std::invalid_argument("Currently, dimensions cannot be >3.");
399  }
400  }
401 }
