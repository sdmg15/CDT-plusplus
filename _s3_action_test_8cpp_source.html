16 SCENARIO("Calculate the bulk action on S3 triangulations", "[action][!mayfail]")
17 {
18  GIVEN("A 3D 2-sphere foliated triangulation.")
19  {
20  constexpr auto simplices = static_cast<std::int_fast32_t>(6400);
21  constexpr auto timeslices = static_cast<std::int_fast32_t>(7);
22  constexpr auto K = static_cast<long double>(1.1);
23  constexpr auto Lambda = static_cast<long double>(0.1);
24  SimplicialManifold universe(simplices, timeslices);
25  // Verify triangulation
26  CHECK(universe.triangulation);
27  CHECK(universe.geometry->number_of_cells() ==
28  universe.triangulation->number_of_finite_cells());
29  CHECK(universe.geometry->number_of_edges() ==
30  universe.triangulation->number_of_finite_edges());
31  CHECK(universe.geometry->N0() ==
32  universe.triangulation->number_of_vertices());
33  CHECK(universe.triangulation->dimension() == 3);
34  CHECK(fix_timeslices(universe.triangulation));
35  CHECK(universe.triangulation->is_valid());
36  CHECK(universe.triangulation->tds().is_valid());
37 
38  VolumePerTimeslice(universe);
39 
40  CHECK(universe.geometry->max_timevalue().get() == timeslices);
41  CHECK(universe.geometry->min_timevalue().get() == 1);
42  WHEN("The alpha=-1 Bulk Action is calculated.")
43  {
44  auto Bulk_action = S3_bulk_action_alpha_minus_one(
45  universe.geometry->N1_TL(), universe.geometry->N3_31_13(),
46  universe.geometry->N3_22(), K, Lambda);
47  THEN("The action falls within accepted values.")
48  {
49  std::cout << "S3_bulk_action_alpha_minus_one() = " << Bulk_action
50  << "\n";
51  REQUIRE(3500 <= Bulk_action);
52  REQUIRE(Bulk_action <= 4500);
53  }
54  }
55  WHEN("The alpha=1 Bulk Action is calculated.")
56  {
57  auto Bulk_action = S3_bulk_action_alpha_one(
58  universe.geometry->N1_TL(), universe.geometry->N3_31_13(),
59  universe.geometry->N3_22(), K, Lambda);
60  THEN("The action falls within accepted values.")
61  {
62  std::cout << "S3_bulk_action_alpha_one() = " << Bulk_action << "\n";
63  REQUIRE(2000 <= Bulk_action);
64  REQUIRE(Bulk_action <= 3000);
65  }
66  }
67  WHEN("The generalized Bulk Action is calculated.")
68  {
69  constexpr auto Alpha = static_cast<long double>(0.6);
70  std::cout << "(Long double) Alpha = " << Alpha << '\n';
71  auto Bulk_action = S3_bulk_action(
72  universe.geometry->N1_TL(), universe.geometry->N3_31_13(),
73  universe.geometry->N3_22(), Alpha, K, Lambda);
74  THEN("The action falls within accepted values.")
75  {
76  std::cout << "S3_bulk_action() = " << Bulk_action << "\n";
77  REQUIRE(2700 <= Bulk_action);
78  REQUIRE(Bulk_action <= 3700);
79  }
80  }
81  WHEN(
82  "S3_bulk_action(alpha=1) and S3_bulk_action_alpha_one() are "
83  "calculated.")
84  {
85  constexpr auto tolerance = static_cast<long double>(0.05);
86  constexpr auto Alpha = static_cast<long double>(1.0);
87  auto Bulk_action = S3_bulk_action(
88  universe.geometry->N1_TL(), universe.geometry->N3_31_13(),
89  universe.geometry->N3_22(), Alpha, K, Lambda);
90  auto Bulk_action_one = S3_bulk_action_alpha_one(
91  universe.geometry->N1_TL(), universe.geometry->N3_31_13(),
92  universe.geometry->N3_22(), K, Lambda);
93  THEN(
94  "S3_bulk_action(alpha=1) == S3_bulk_action_alpha_one() within "
95  "tolerances.")
96  {
97  std::cout << "S3_bulk_action() = " << Bulk_action << "\n";
98  std::cout << "S3_bulk_action_alpha_one() = " << Bulk_action_one << "\n";
99  const auto min = Bulk_action_one * (1.0 - tolerance);
100  const auto max = Bulk_action_one * (1.0 + tolerance);
101  REQUIRE(min <= Bulk_action);
102  REQUIRE(Bulk_action <= max);
103  }
104  }
