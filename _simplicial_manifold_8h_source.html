52 /// @brief Classifies edges
53 ///
54 /// This function iterates over all edges in the triangulation
55 /// and classifies them as timelike or spacelike.
56 /// Timelike edges are stored in the **timelike_edges** vector as an Edge_handle
57 /// (tuple of Cell_handle, std::int_fast32_t, std::int_fast32_t) for later use by
58 /// ergodic moves on timelike edges. Spacelike edges are also stored as a
59 /// vector of Edge_handle **spacelike_edges**, for use by (4,4) moves as
60 /// well as the distance-finding algorithms.
61 /// @param[in] universe_ptr A std::unique_ptr<Delaunay> to the triangulation
62 /// @returns A std::pair<std::vector<Edge_handle>, std::vector<Edge_handle>> of
63 /// timelike edges and spacelike edges
64 template <typename T>
65 auto classify_edges(T&& universe_ptr)
66 {
67 #ifndef NDEBUG
68  std::cout << "Classifying edges....\n";
69 #endif
70  Delaunay::Finite_edges_iterator eit;
71  std::vector<Edge_handle> timelike_edges;
72  std::vector<Edge_handle> spacelike_edges;
73 
74  // Iterate over all edges in the Delaunay triangulation
75  for (eit = universe_ptr->finite_edges_begin();
76  eit != universe_ptr->finite_edges_end(); ++eit)
77  {
78  Cell_handle ch = eit->first;
79  // Get timevalues of vertices at the edge ends
80  auto time1 = ch->vertex(eit->second)->info();
81  auto time2 = ch->vertex(eit->third)->info();
82 
83  // Make Edge_handle
84  Edge_handle thisEdge{
85  ch, static_cast<std::int_fast32_t>(ch->index(ch->vertex(eit->second))),
86  static_cast<std::int_fast32_t>(ch->index(ch->vertex(eit->third)))};
87 
88  if (time1 != time2) { // We have a timelike edge
89  timelike_edges.emplace_back(thisEdge);
90 
91 #ifdef DETAILED_DEBUGGING
92  std::cout << "First vertex of edge is " << std::get<1>(thisEdge)
93  << " and second vertex of edge is " << std::get<2>(thisEdge)
94  << std::endl;
95 #endif
96  }
97  else
98  { // We have a spacelike edge
99  spacelike_edges.emplace_back(thisEdge);
100  } // endif
101  } // Finish iterating over edges
102 
103 // Display results if debugging
104 #ifndef NDEBUG
105  std::cout << "There are " << timelike_edges.size() << " timelike edges and "
106  << spacelike_edges.size() << " spacelike edges.\n";
107 #endif
108  return std::make_pair(timelike_edges, spacelike_edges);
109 } // classify_edges()

111 /// @brief Classify simplices as (3,1), (2,2), or (1,3)
112 ///
113 /// This function iterates over all cells in the triangulation
114 /// and classifies them as:
115 /// \f{eqnarray*}{
116 /// 31 &=& (3, 1) \\
117 /// 22 &=& (2, 2) \\
118 /// 13 &=& (1, 3)
119 /// \f}
120 /// The vectors **three_one**, **two_two**, and **one_three** contain
121 /// Cell_handles to all the simplices in the triangulation of that corresponding
122 /// type.
123 ///
124 /// @param[in] universe_ptr A std::unique_ptr<Delaunay> to the triangulation
125 /// @returns A std::tuple<std::vector, std::vector, std::vector> of
126 /// **three_one**, **two_two**, and **one_three**
127 template <typename T>
128 auto classify_simplices(T&& universe_ptr)
129 {
130 #ifndef NDEBUG
131  std::cout << "Classifying simplices....\n";
132 #endif
133  Delaunay::Finite_cells_iterator cit;
134  std::vector<Cell_handle> three_one;
135  std::vector<Cell_handle> two_two;
136  std::vector<Cell_handle> one_three;
137 
138  // Iterate over all cells in the Delaunay triangulation
139  for (cit = universe_ptr->finite_cells_begin();
140  cit != universe_ptr->finite_cells_end(); ++cit)
141  {
142  std::int_fast32_t max_values{0};
143  std::int_fast32_t min_values{0};
144  // Push every time value of every vertex into a list
145  std::int_fast32_t timevalues[4] = {
146  cit->vertex(0)->info(),
147  cit->vertex(1)->info(),
148  cit->vertex(2)->info(),
149  cit->vertex(3)->info(),
150  };
151  std::int_fast32_t max_time =
152  *std::max_element(std::begin(timevalues), std::end(timevalues));
153  for (auto elt : timevalues) {
154  if (elt == max_time) {
155  ++max_values;
156  }
157  else
158  {
159  ++min_values;
160  }
161  }
162 
163  // Classify simplex using max_values and write to cit->info()
164  if (min_values == 1 && max_values == 3) {
165  cit->info() = 13;
166  one_three.emplace_back(cit);
167  }
168  else if (min_values == 2 && max_values == 2)
169  {
170  cit->info() = 22;
171  two_two.emplace_back(cit);
172  }
173  else if (min_values == 3 && max_values == 1)
174  {
175  cit->info() = 31;
176  three_one.emplace_back(cit);
177  }
178  else
179  {
180  throw std::runtime_error("Invalid simplex in classify_simplices()!");
181  } // endif
182  } // Finish iterating over cells
183 
184 // Display results if debugging
185 #ifndef NDEBUG
186  std::cout << "There are " << three_one.size() << " (3,1) simplices and "
187  << two_two.size() << " (2,2) simplices\n";
188  std::cout << "and " << one_three.size() << " (1,3) simplices.\n";
189 #endif
190  return std::make_tuple(three_one, two_two, one_three);
191 } // classify_simplices()

193 template <typename T>
194 auto classify_all_simplices(T&& universe_ptr)
195 {
196 #ifndef NDEBUG
197  std::cout << "Classifying all simplices....\n";
198 #endif
199 
200  auto cells = classify_simplices(universe_ptr);
201  auto edges = classify_edges(universe_ptr);
202  std::vector<Vertex_handle> vertices;
203  for (auto vit = universe_ptr->finite_vertices_begin();
204  vit != universe_ptr->finite_vertices_end(); ++vit)
205  {
206  vertices.emplace_back(vit);
207  }
208  return std::make_tuple(std::get<0>(cells), std::get<1>(cells),
209  std::get<2>(cells), edges.first, edges.second,
210  vertices);
211 }
