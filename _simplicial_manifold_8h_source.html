48 auto classify_edges(T&& universe_ptr)
49 {
50 #ifndef NDEBUG
51  std::cout << "Classifying edges....\n";
52 #endif
53  Delaunay::Finite_edges_iterator eit;
54  std::vector<Edge_handle> timelike_edges;
55  std::vector<Edge_handle> spacelike_edges;
56 
57  // Iterate over all edges in the Delaunay triangulation
58  for (eit = universe_ptr->finite_edges_begin();
59  eit != universe_ptr->finite_edges_end(); ++eit)
60  {
61  Cell_handle ch = eit->first;
62  // Get timevalues of vertices at the edge ends
63  auto time1 = ch->vertex(eit->second)->info();
64  auto time2 = ch->vertex(eit->third)->info();
65 
66  // Make Edge_handle
67  Edge_handle thisEdge{
68  ch, static_cast<std::intmax_t>(ch->index(ch->vertex(eit->second))),
69  static_cast<std::intmax_t>(ch->index(ch->vertex(eit->third)))};
70 
71  if (time1 != time2) { // We have a timelike edge
72  timelike_edges.emplace_back(thisEdge);
73 
74 #ifdef DETAILED_DEBUGGING
75  std::cout << "First vertex of edge is " << std::get<1>(thisEdge)
76  << " and second vertex of edge is " << std::get<2>(thisEdge)
77  << std::endl;
78 #endif
79  }
80  else
81  { // We have a spacelike edge
82  spacelike_edges.emplace_back(thisEdge);
83  } // endif
84  } // Finish iterating over edges
85 
86 // Display results if debugging
87 #ifndef NDEBUG
88  std::cout << "There are " << timelike_edges.size() << " timelike edges and "
89  << spacelike_edges.size() << " spacelike edges.\n";
90 #endif
91  return std::make_pair(timelike_edges, spacelike_edges);
92 } // classify_edges()

110 template <typename T>
111 auto classify_simplices(T&& universe_ptr)
112 {
113 #ifndef NDEBUG
114  std::cout << "Classifying simplices....\n";
115 #endif
116  Delaunay::Finite_cells_iterator cit;
117  std::vector<Cell_handle> three_one;
118  std::vector<Cell_handle> two_two;
119  std::vector<Cell_handle> one_three;
120 
121  // Iterate over all cells in the Delaunay triangulation
122  for (cit = universe_ptr->finite_cells_begin();
123  cit != universe_ptr->finite_cells_end(); ++cit)
124  {
125  std::intmax_t max_values{0};
126  std::intmax_t min_values{0};
127  // Push every time value of every vertex into a list
128  std::intmax_t timevalues[4] = {
129  cit->vertex(0)->info(),
130  cit->vertex(1)->info(),
131  cit->vertex(2)->info(),
132  cit->vertex(3)->info(),
133  };
134  std::intmax_t max_time =
135  *std::max_element(std::begin(timevalues), std::end(timevalues));
136  for (auto elt : timevalues) {
137  if (elt == max_time) {
138  ++max_values;
139  }
140  else
141  {
142  ++min_values;
143  }
144  }
145 
146  // Classify simplex using max_values and write to cit->info()
147  if (min_values == 1 && max_values == 3) {
148  cit->info() = 13;
149  one_three.emplace_back(cit);
150  }
151  else if (min_values == 2 && max_values == 2)
152  {
153  cit->info() = 22;
154  two_two.emplace_back(cit);
155  }
156  else if (min_values == 3 && max_values == 1)
157  {
158  cit->info() = 31;
159  three_one.emplace_back(cit);
160  }
161  else
162  {
163  throw std::runtime_error("Invalid simplex in classify_simplices()!");
164  } // endif
165  } // Finish iterating over cells
166 
167 // Display results if debugging
168 #ifndef NDEBUG
169  std::cout << "There are " << three_one.size() << " (3,1) simplices and "
170  << two_two.size() << " (2,2) simplices\n";
171  std::cout << "and " << one_three.size() << " (1,3) simplices.\n";
172 #endif
173  return std::make_tuple(three_one, two_two, one_three);
174 } // classify_simplices()

176 template <typename T>
177 auto classify_all_simplices(T&& universe_ptr)
178 {
179 #ifndef NDEBUG

181 #endif
182 
183  auto cells = classify_simplices(universe_ptr);
184  auto edges = classify_edges(universe_ptr);
185  std::vector<Vertex_handle> vertices;
186  for (auto vit = universe_ptr->finite_vertices_begin();

189  vertices.emplace_back(vit);
190  }
191  return std::make_tuple(std::get<0>(cells), std::get<1>(cells),
192  std::get<2>(cells), edges.first, edges.second,
193  vertices);
