52 /// @brief Classifies edges
53 ///
54 /// This function iterates over all edges in the triangulation
55 /// and classifies them as timelike or spacelike.
56 /// Timelike edges are stored in the **timelike_edges** vector as an Edge_handle
57 /// (tuple of Cell_handle, std::int32_t, std::int32_t) for later use by
58 /// ergodic moves on timelike edges. Spacelike edges are also stored as a
59 /// vector of Edge_handle **spacelike_edges**, for use by (4,4) moves as
60 /// well as the distance-finding algorithms.
61 /// @param[in] universe_ptr A std::unique_ptr<Delaunay> to the triangulation
62 /// @returns A std::pair<std::vector<Edge_handle>, std::vector<Edge_handle>> of
63 /// timelike edges and spacelike edges
64 template <typename T>
65 auto classify_edges(T&& universe_ptr)
66 {
67 #ifndef NDEBUG
68  std::cout << "Classifying edges....\n";
69 #endif
70  Delaunay::Finite_edges_iterator eit;
71  std::vector<Edge_handle> timelike_edges;
72  std::vector<Edge_handle> spacelike_edges;
73 
74  // Iterate over all edges in the Delaunay triangulation
75  for (eit = universe_ptr->finite_edges_begin();
76  eit != universe_ptr->finite_edges_end(); ++eit)
77  {
78  Cell_handle ch = eit->first;
79  // Get timevalues of vertices at the edge ends
80  auto time1 = ch->vertex(eit->second)->info();
81  auto time2 = ch->vertex(eit->third)->info();
82 
83  // Make Edge_handle
84  Edge_handle thisEdge{
85  ch, static_cast<std::int32_t>(ch->index(ch->vertex(eit->second))),
86  static_cast<std::int32_t>(ch->index(ch->vertex(eit->third)))};
87 
88  if (time1 != time2)
89  { // We have a timelike edge
90  timelike_edges.emplace_back(thisEdge);
91 
92 #ifdef DETAILED_DEBUGGING
93  std::cout << "First vertex of edge is " << std::get<1>(thisEdge)
94  << " and second vertex of edge is " << std::get<2>(thisEdge)
95  << std::endl;
96 #endif
97  }
98  else
99  { // We have a spacelike edge
100  spacelike_edges.emplace_back(thisEdge);
101  } // endif
102  } // Finish iterating over edges
103 
104 // Display results if debugging
105 #ifndef NDEBUG
106  std::cout << "There are " << timelike_edges.size() << " timelike edges and "
107  << spacelike_edges.size() << " spacelike edges.\n";
108 #endif
109  return std::make_pair(timelike_edges, spacelike_edges);
110 } // classify_edges()

112 /// @brief Classify simplices as (3,1), (2,2), or (1,3)
113 ///
114 /// This function iterates over all cells in the triangulation
115 /// and classifies them as:
116 /// \f{eqnarray*}{
117 /// 31 &=& (3, 1) \\
118 /// 22 &=& (2, 2) \\
119 /// 13 &=& (1, 3)
120 /// \f}
121 /// The vectors **three_one**, **two_two**, and **one_three** contain
122 /// Cell_handles to all the simplices in the triangulation of that corresponding
123 /// type.
124 ///
125 /// @param[in] universe_ptr A std::unique_ptr<Delaunay> to the triangulation
126 /// @returns A std::tuple<std::vector, std::vector, std::vector> of
127 /// **three_one**, **two_two**, and **one_three**
128 template <typename T>
129 auto classify_simplices(T&& universe_ptr)
130 {
131 #ifndef NDEBUG
132  std::cout << "Classifying simplices....\n";
133 #endif
134  Delaunay::Finite_cells_iterator cit;
135  std::vector<Cell_handle> three_one;
136  std::vector<Cell_handle> two_two;
137  std::vector<Cell_handle> one_three;
138 
139  // Iterate over all cells in the Delaunay triangulation
140  for (cit = universe_ptr->finite_cells_begin();
141  cit != universe_ptr->finite_cells_end(); ++cit)
142  {
143  std::int32_t max_values{0};
144  std::int32_t min_values{0};
145  // Push every time value of every vertex into a list
146  std::int32_t timevalues[4] = {
147  cit->vertex(0)->info(),
148  cit->vertex(1)->info(),
149  cit->vertex(2)->info(),
150  cit->vertex(3)->info(),
151  };
152  std::int32_t max_time =
153  *std::max_element(std::begin(timevalues), std::end(timevalues));
154  for (auto elt : timevalues)
155  {
156  if (elt == max_time) { ++max_values; }
157  else
158  {
159  ++min_values;
160  }
161  }
162 
163  // Classify simplex using max_values and write to cit->info()
164  if (min_values == 1 && max_values == 3)
165  {
166  cit->info() = 13;
167  one_three.emplace_back(cit);
168  }
169  else if (min_values == 2 && max_values == 2)
170  {
171  cit->info() = 22;
172  two_two.emplace_back(cit);
173  }
174  else if (min_values == 3 && max_values == 1)
175  {
176  cit->info() = 31;
177  three_one.emplace_back(cit);
178  }
179  else
180  {
181  throw std::runtime_error("Invalid simplex in classify_simplices()!");
182  } // endif
183  } // Finish iterating over cells
184 
185 // Display results if debugging
186 #ifndef NDEBUG
187  std::cout << "There are " << three_one.size() << " (3,1) simplices and "
188  << two_two.size() << " (2,2) simplices\n";
189  std::cout << "and " << one_three.size() << " (1,3) simplices.\n";
190 #endif
191  return std::make_tuple(three_one, two_two, one_three);
192 } // classify_simplices()

194 template <typename T>
195 auto classify_all_simplices(T&& universe_ptr)
196 {
197 #ifndef NDEBUG
198  std::cout << "Classifying all simplices....\n";
199 #endif
200 
201  auto cells = classify_simplices(universe_ptr);
202  auto edges = classify_edges(universe_ptr);
203  std::vector<Vertex_handle> vertices;
204  for (auto vit = universe_ptr->finite_vertices_begin();
205  vit != universe_ptr->finite_vertices_end(); ++vit)
206  { vertices.emplace_back(vit); }
207  return std::make_tuple(std::get<0>(cells), std::get<1>(cells),
208  std::get<2>(cells), edges.first, edges.second,
209  vertices);
210 }
