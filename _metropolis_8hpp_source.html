45 /// @brief Metropolis-Hastings algorithm function object

47 /// The Metropolis-Hastings algorithm is a Markov Chain Monte Carlo method.
48 /// The probability of making an ergodic (Pachner) move is:
49 ///
50 /// \f[P_{ergodic move}=a_{1}a_{2}\f]
51 /// \f[a_1=\frac{move[i]}{\sum\limits_{i}move[i]}\f]
52 /// \f[a_2=e^{\Delta S}\f]

120  /// \f[a_1=\frac{move[i]}{\sum\limits_{i}move[i]}\f]

169  auto newS3Action = static_cast<Gmpzf>(0);

171  switch (move)
172  {
173  case move_type::TWO_THREE:
174  // A (2,3) move adds a timelike edge

177  K_, Lambda_);
178  break;
179  case move_type::THREE_TWO:
180  // A (3,2) move removes a timelike edge
181  // and a (2,2) simplex
182  newS3Action = S3_bulk_action(N1_TL_ - 1, N3_31_13_, N3_22_ - 1, Alpha_,
183  K_, Lambda_);
184  break;
185  case move_type::TWO_SIX:
186  // A (2,6) move adds 2 timelike edges and
187  // 2 (1,3) and 2 (3,1) simplices
188  newS3Action = S3_bulk_action(N1_TL_ + 2, N3_31_13_ + 4, N3_22_, Alpha_,
189  K_, Lambda_);
190  break;
191  case move_type::SIX_TWO:
192  // A (6,2) move removes 2 timelike edges and
193  // 2 (1,3) and 2 (3,1) simplices
194  newS3Action = S3_bulk_action(N1_TL_ - 2, N3_31_13_, N3_22_ - 4, Alpha_,
195  K_, Lambda_);
196  break;
197  case move_type::FOUR_FOUR:
198 // A (4,4) move changes nothing with respect to the action,
199 // and e^0==1

241  /// @brief Attempt a move of the selected type
242  ///
243  /// This function implements the core of the Metropolis-Hastings algorithm
244  /// by generating a random number and comparing with the results of
245  /// CalculateA1 and CalculateA2. If the move is accepted, this function
246  /// calls make_move(). If not, it updates **attempted_moves_**.
247  ///
248  /// @param move The type of move
249  void attempt_move(const move_type move)
250  {
251  // Calculate probability
252  auto a1 = CalculateA1(move);
253  // Make move if random number < probability
254  auto a2 = CalculateA2(move);
255 
256  const auto trial_value = generate_probability();
257  // Convert to Gmpzf because trial_value will be set to 0 when
258  // comparing with a1 and a2!
259  // const auto trial = Gmpzf(static_cast<double>(trial_value));
260  const auto trial = static_cast<double>(trial_value);
261 
262 #ifndef NDEBUG
263  std::cout << __PRETTY_FUNCTION__ << " called.\n";
264  std::cout << "trial_value = " << trial_value << "\n";
265  std::cout << "trial = " << trial << "\n";
266 #endif
267 
268  if (trial <= a1 * a2)
269  {
270  // Move accepted
271  make_move(move);
272  }
273  else
274  {
275  // Move rejected
276  // Increment attempted_moves_
277  ++attempted_moves_[to_integral(move)];
278  }
279 
280 #ifndef NDEBUG

282  std::cout << "Attempting move.\n";

284  std::cout << "Trial = " << trial << "\n";

286  std::cout << "A2 = " << a2 << "\n";

288  std::cout << ((trial <= a1 * a2) ? "Move accepted." : "Move rejected.")
289  << "\n";
290  print_run();
291 #endif
292  } // attempt_move()
