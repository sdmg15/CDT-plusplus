103 /// @brief Fix simplices with incorrect foliation
104 ///
105 /// This function iterates over all of the cells in the triangulation.
106 /// Within each cell, it iterates over all of the vertices and reads timeslices.
107 /// Validity of the cell is first checked by the **is_valid()** function.
108 /// The foliation validity is then checked by finding maximum and minimum
109 /// timeslices for all the vertices of a cell and ensuring that the difference
110 /// is exactly 1.
111 /// If a cell has a bad foliation, the vertex with the highest timeslice is
112 /// deleted. The Delaunay triangulation is then recomputed on the remaining
113 /// vertices.
114 /// This function is repeatedly called by fix_triangulation() up to
115 /// **MAX_FOLIATION_FIX_PASSES** times.
116 ///
117 /// @param universe_ptr A std::unique_ptr<Delaunay> to the triangulation
118 /// @returns A boolean value if there are invalid simplices
119 template <typename T>
120 auto fix_timeslices(T&& universe_ptr)
121 {
122  // Delaunay::Finite_cells_iterator cit;
123  std::int32_t min_time{0};
124  std::int32_t max_time{0};
125  std::int32_t valid{0};
126  std::int32_t invalid{0};
127  std::int32_t max_vertex{0};
128  std::set<Vertex_handle> deleted_vertices;
129 
130  // Iterate over all cells in the Delaunay triangulation
131  for (Delaunay3::Finite_cells_iterator cit =
132  universe_ptr->finite_cells_begin();
133  cit != universe_ptr->finite_cells_end(); ++cit)
134  {
135  if (cit->is_valid())
136  { // Valid cell
137  min_time = cit->vertex(0)->info();
138  max_time = min_time;
139 #ifdef DETAILED_DEBUGGING
140  bool this_cell_foliation_valid = true;
141 #endif
142  // Iterate over all vertices in the cell
143  for (auto i = 0; i < 4; ++i)
144  {
145  auto current_time = cit->vertex(i)->info();
146 
147  // Classify extreme values
148  if (current_time < min_time) min_time = current_time;
149  if (current_time > max_time)
150  {
151  max_time = current_time;
152  max_vertex = static_cast<int32_t>(i);
153  }
154  } // Finish iterating over vertices
155  // There should be a difference of 1 between min_time and max_time
156  if (max_time - min_time != 1)
157  {
158  invalid++;
159 #ifdef DETAILED_DEBUGGING
160  this_cell_foliation_valid = false;
161 #endif
162  // Single-threaded delete max vertex
163  // universe_ptr->remove(cit->vertex(max_vertex));
164 
165  // Parallel delete std::set of max_vertex for all invalid cells
166  deleted_vertices.emplace(cit->vertex(static_cast<int>(max_vertex)));
167  }
168  else
169  {
170  ++valid;
171  }
172 
173 #ifdef DETAILED_DEBUGGING
174  std::cout << "Foliation for cell is "
175  << ((this_cell_foliation_valid) ? "valid." : "invalid.\n");
176  for (auto i = 0; i < 4; ++i)
177  {
178  std::cout << "Vertex " << i << " is " << cit->vertex(i)->point()
179  << " with timeslice " << cit->vertex(i)->info() << "\n";
180  }
181 #endif
182  }
183  else
184  {
185  throw std::runtime_error("Cell handle is invalid!");
186  // Or just remove the cell
187  // universe_ptr->tds().delete_cell(cit);
188  // This results in a possibly broken Delaunay triangulation
189  // Or possibly just delete a vertex in the cell,
190  // perhaps forcing a re-triangulation?
191  }
192  } // Finish iterating over cells
193 
194  // Delete invalid vertices
195  universe_ptr->remove(deleted_vertices.begin(), deleted_vertices.end());
196  // Check that the triangulation is still valid
197  // Turned off by -DCGAL_TRIANGULATION_NO_POSTCONDITIONS
198  // CGAL_triangulation_expensive_postcondition(universe_ptr->is_valid());
199  if (!universe_ptr->tds().is_valid())
200  throw std::logic_error("Delaunay tds invalid!");
201 
202 #ifndef NDEBUG
203  std::cout << "There are " << invalid << " invalid simplices and " << valid
204  << " valid simplices.\n";
205 #endif
206  return invalid == 0;
207 } // fix_timeslices

209 /// @brief Fixes the foliation of the triangulation
210 ///
211 /// Runs fix_timeslices() to fix foliation until there are no errors,
212 /// or MAX_FOLIATION_FIX_PASSES whichever comes first.
213 ///
214 /// @param universe_ptr A std::unique_ptr<Delaunay> to the triangulation
215 template <typename T>
216 void fix_triangulation(T&& universe_ptr)
217 {
218  for (std::int32_t pass = 0; pass < MAX_FOLIATION_FIX_PASSES; ++pass)
219  {
220 #ifndef NDEBUG
221  std::cout << "Fix Pass #" << (pass + 1) << "\n";
222 #endif
223  if (fix_timeslices(universe_ptr)) break;
224  }
225  if (!fix_timeslices(universe_ptr))
226  throw std::logic_error("Delaunay triangulation not correctly foliated.");
227 } // fix_triangulation()

240 /// @brief Make foliated spheres
241 ///
242 /// The radius is used to denote the time value, so we can nest 2-spheres
243 /// such that our time foliation contains leaves of identical topology.
244 ///
245 /// @param simplices The number of desired simplices in the triangulation
246 /// @param timeslices The number of desired timeslices in the triangulation
247 /// @return A std::vector<std::pair<Point, std::int32_t>> containing random
248 /// vertices and their corresponding timevalues
249 auto inline make_foliated_sphere(const std::int32_t simplices,
250  const std::int32_t timeslices,
251  double initial_radius = INITIAL_RADIUS,
252  double radial_factor = RADIAL_FACTOR)
253 {
254  // double radius{0};
255  const auto points_per_timeslice =
256  expected_points_per_simplex(DIMENSION, simplices, timeslices);
257  CGAL_triangulation_precondition(points_per_timeslice >= 2);
258  Causal_vertices causal_vertices;
259  using Spherical_points_generator_3 = CGAL::Random_points_on_sphere_3<Point>;
260 
261  for (std::int32_t i = 0; i < timeslices; ++i)
262  {
263  auto radius = initial_radius + static_cast<double>(i) * radial_factor;
264  // CGAL::Random_points_on_sphere_3<Point> gen{radius};
265  Spherical_points_generator_3 gen{radius};
266  // At each radius, generate a sphere of random points proportional to area
267  for (std::int32_t j = 0;
268  j < static_cast<std::int32_t>(points_per_timeslice * radius); ++j)
269  { causal_vertices.emplace_back(std::make_pair(*gen++, i + 1)); } // end j
270  } // end i
271  return causal_vertices;
272 } // make_foliated_sphere()

274 /// @brief Make a triangulation from foliated 2-spheres
275 ///
276 /// This function creates a triangulation from successive spheres.
277 /// First, the number of points per leaf in the foliation is estimated given
278 /// the desired number of simplices.
279 /// Next, make_foliated_sphere() is called to generate nested spheres.
280 /// The radius of the sphere is assigned as the time value for each vertex
281 /// in that sphere, which comprises a leaf in the foliation.
282 /// All vertices in all spheres (along with their time values) are then
283 /// inserted with insert_into_triangulation() into a Delaunay triangulation
284 /// (see http://en.wikipedia.org/wiki/Delaunay_triangulation for details).
285 /// Finally, fix_triangulation() removes cells in the Delaunay triangulation
286 /// with invalid foliations using fix_timeslices(). A last check is performed
287 /// to ensure a valid Delaunay triangulation.
288 ///
289 /// @param[in] simplices The number of desired simplices in the triangulation
290 /// @param[in] timeslices The number of timeslices in the triangulation
291 /// @returns A std::unique_ptr<Delaunay> to the foliated triangulation
292 auto inline make_triangulation(const std::int32_t simplices,
293  const std::int32_t timeslices,
294  double initial_radius = INITIAL_RADIUS,
295  double radial_factor = RADIAL_FACTOR)
296 {
297  std::cout << "Generating universe ... \n";
298 
299 #ifdef CGAL_LINKED_WITH_TBB
300  // Construct the locking data-structure
301  // using the bounding-box of the points
302  auto bounding_box_size = static_cast<double>(timeslices + 1);
303  Delaunay::Lock_data_structure locking_ds{
304  CGAL::Bbox_3{-bounding_box_size, -bounding_box_size, -bounding_box_size,
305  bounding_box_size, bounding_box_size, bounding_box_size},
306  50};
307  Delaunay universe{K{}, &locking_ds};
308 #else
309  Delaunay3 universe{};
310 #endif
311 
312  auto universe_ptr = std::make_unique<decltype(universe)>(universe);
313  auto causal_vertices = make_foliated_sphere(simplices, timeslices,
314  initial_radius, radial_factor);
315  insert_into_triangulation(universe_ptr, causal_vertices);
316  fix_triangulation(universe_ptr);
317  if (!universe_ptr->is_valid())
318  throw std::logic_error("Delaunay triangulation invalid!");
319  return universe_ptr;
320 } // make_triangulation()
