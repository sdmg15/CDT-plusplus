282  Lambda_);
283  break;

287  newS3Action = S3_bulk_action(N1_TL_+2,

289  N3_22_,

291  K_,
292  Lambda_);
293  break;

297  newS3Action = S3_bulk_action(N1_TL_-2,
298  N3_31_,
299  N3_22_-4,
300  Alpha_,
301  K_,
302  Lambda_);
