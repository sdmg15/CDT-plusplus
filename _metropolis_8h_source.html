269  auto CalculateA2(move_type move) const noexcept {
270  auto currentS3Action =
271  S3_bulk_action(N1_TL_, N3_31_, N3_22_, Alpha_, K_, Lambda_);
272  auto newS3Action = static_cast<Gmpzf>(0);
273  // auto newS3Action = static_cast<MP_Float>(0);
274  switch (move) {
275  case move_type::TWO_THREE:
276  // A (2,3) move adds a timelike edge
277  // and a (2,2) simplex
278  newS3Action =
279  S3_bulk_action(N1_TL_ + 1, N3_31_, N3_22_ + 1, Alpha_, K_, Lambda_);
280  break;
281  case move_type::THREE_TWO:
282  // A (3,2) move removes a timelike edge
283  // and a (2,2) simplex
284  newS3Action =
285  S3_bulk_action(N1_TL_ - 1, N3_31_, N3_22_ - 1, Alpha_, K_, Lambda_);
286  break;
287  case move_type::TWO_SIX:
288  // A (2,6) move adds 2 timelike edges and
289  // 2 (1,3) and 2 (3,1) simplices
290  newS3Action =
291  S3_bulk_action(N1_TL_ + 2, N3_31_ + 4, N3_22_, Alpha_, K_, Lambda_);
292  break;
293  case move_type::SIX_TWO:
294  // A (6,2) move removes 2 timelike edges and
295  // 2 (1,3) and 2 (3,1) simplices
296  newS3Action =
297  S3_bulk_action(N1_TL_ - 2, N3_31_, N3_22_ - 4, Alpha_, K_, Lambda_);
298  break;
299  case move_type::FOUR_FOUR:
300 // A (4,4) move changes nothing with respect to the action,
301 // and e^0==1
302 #ifndef NDEBUG
303  std::cout << "A2 is 1" << std::endl;
304 #endif
305  return static_cast<Gmpzf>(1);
306  default:
307  assert(!"Metropolis::CalculateA2 should never get here!");
308  }
309 
310  auto exponent = newS3Action - currentS3Action;
311  auto exponent_double = Gmpzf_to_double(exponent);
312 
313  // if exponent > 0 then e^exponent >=1 so according to Metropolis
314  // algorithm return A2=1
315  if (exponent >= 0) return static_cast<Gmpzf>(1);
316 
317  // Set precision for initialization and assignment functions
318  mpfr_set_default_prec(PRECISION);
319 
320  // Initialize for MPFR
321  mpfr_t r1, a2;
322  mpfr_inits2(PRECISION, r1, a2, nullptr);
323 
324  // Set input parameters and constants to mpfr_t equivalents
325  mpfr_init_set_d(r1, exponent_double, MPFR_RNDD); // r1 = exponent
326 
327  // e^exponent

329 
330  // Convert mpfr_t total to Gmpzf result by using Gmpzf(double d)
