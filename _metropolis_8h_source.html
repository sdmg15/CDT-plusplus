284  auto CalculateA2(move_type move) const noexcept {
285  auto currentS3Action = S3_bulk_action(N1_TL_.load(),
286  N3_31_.load(),
287  N3_22_.load(),
288  Alpha_,
289  K_,
290  Lambda_);
291  auto newS3Action = static_cast<Gmpzf>(0);
292  // auto newS3Action = static_cast<MP_Float>(0);
293  switch (move) {
294  case move_type::TWO_THREE:
295  // A (2,3) move adds a timelike edge
296  // and a (2,2) simplex
297  newS3Action = S3_bulk_action(N1_TL_.load()+1,
298  N3_31_.load(),
299  N3_22_.load()+1,
300  Alpha_,
301  K_,
302  Lambda_);
303  break;
304  case move_type::THREE_TWO:
305  // A (3,2) move removes a timelike edge
306  // and a (2,2) simplex
307  newS3Action = S3_bulk_action(N1_TL_.load()-1,
308  N3_31_.load(),
309  N3_22_.load()-1,
310  Alpha_,
311  K_,
312  Lambda_);
313  break;
314  case move_type::TWO_SIX:
315  // A (2,6) move adds 2 timelike edges and
316  // 2 (1,3) and 2 (3,1) simplices
317  newS3Action = S3_bulk_action(N1_TL_.load()+2,
318  N3_31_.load()+4,
319  N3_22_.load(),
320  Alpha_,
321  K_,
322  Lambda_);
323  break;
324  case move_type::SIX_TWO:
325  // A (6,2) move removes 2 timelike edges and
326  // 2 (1,3) and 2 (3,1) simplices
327  newS3Action = S3_bulk_action(N1_TL_.load()-2,
328  N3_31_.load(),
329  N3_22_.load()-4,
330  Alpha_,
331  K_,
332  Lambda_);
333  break;
334  case move_type::FOUR_FOUR:
335  // A (4,4) move changes nothing with respect to the action,
336  // and e^0==1

338  std::cout << "A2 is 1" << std::endl;
339  #endif
340  return static_cast<Gmpzf>(1);
341  }
