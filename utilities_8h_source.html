296  const auto simplices_per_timeslice = simplices / timeslices;
297  switch (dimension) {
298  case 3: {
299  // Avoid segfaults for small values
300  if (simplices == timeslices) {
301  return 4 * simplices_per_timeslice;
302  } else if (simplices < 10000) {

304  } else if (simplices < 100000) {
305  return static_cast<std::uintmax_t>(1.5 * simplices_per_timeslice);
306  } else {
