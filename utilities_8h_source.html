300  const auto simplices_per_timeslice = simplices / timeslices;
301  switch (dimension) {
302  case 3: {
303  // Avoid segfaults for small values
304  if (simplices == timeslices) {
305  return 4 * simplices_per_timeslice;
306  } else if (simplices < 10000) {

309  return static_cast<std::uintmax_t>(1.5 * simplices_per_timeslice);
