311  // This is simply not sufficient. Need to check *both* max_time and
312  // min_time, and that there are exactly 1 and 3, 2 and 2, or 3 and 1.
313  // Anything else means we have an invalid simplex which we should
314  // also return.
315  // We also need to check that max_time - min_time = 1, else we have
316  // a mis-classified vertex (probably)
317  auto max_time =
318  std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
319  auto max_time_vertices =
320  std::count_if(cell_vertices.begin(), cell_vertices.end(),
321  [max_time](auto const& vertex) {
322  return vertex->info() == *max_time;
323  });
324  // Check max_time - min_time here
325  switch (max_time_vertices)
326  {
327  case 1:
328  c->info() = static_cast<int>(Cell_type::THREE_ONE);
329  break;
330  case 2:
331  c->info() = static_cast<int>(Cell_type::TWO_TWO);
332  break;
333  case 3:
334  c->info() = static_cast<int>(Cell_type::ONE_THREE);
335  break;
336  default:
337  throw std::logic_error("Mis-classified cell.");
338  }
339  if (debugging)
340  {
341  std::cout << "Max timevalue is " << *max_time << "\n";
342  std::cout << "There are " << max_time_vertices
343  << " vertices with max timeslice in the cell.\n";
344  std::cout << "Cell info is now " << c->info() << "\n";
345  std::cout << "---\n";
346  }
347  cell_vertices.clear();
348  vertex_timevalues.clear();
349  }
