227  [[nodiscard]] auto classify_cells(std::vector<Cell_handle> const& cells,
228  bool debugging = false) const
229  -> std::vector<Cell_handle>
230  {
231  Expects(cells.size() == number_of_cells_);
232  std::vector<Vertex_handle> cell_vertices;
233  cell_vertices.reserve(4);
234  std::vector<int> vertex_timevalues;
235  vertex_timevalues.reserve(4);
236  for (auto const& c : cells)
237  {
238  if (debugging) { std::cout << "Cell info was " << c->info() << '\n'; }
239 
240  for (int j = 0; j < 4; ++j)
241  {
242  cell_vertices.emplace_back(c->vertex(j));
243  vertex_timevalues.emplace_back(c->vertex(j)->info());
244  if (debugging)
245  {
246  std::cout << "Cell vertex " << j << " has timevalue "
247  << c->vertex(j)->info() << '\n';
248  }
249  }
250 
251  auto max_time =
252  std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
253  auto max_time_vertices =
254  std::count_if(cell_vertices.begin(), cell_vertices.end(),
255  [max_time](auto const& vertex) {
256  return vertex->info() == *max_time;
257  });
258 
259  switch (max_time_vertices)
260  {
261  case 1:
262  c->info() = static_cast<int>(Cell_type::ONE_THREE);
263  break;
264  case 2:
265  c->info() = static_cast<int>(Cell_type::TWO_TWO);
266  break;
267  case 3:
268  c->info() = static_cast<int>(Cell_type::THREE_ONE);
269  break;
270  default:
271  throw std::logic_error("Mis-classified cell.");
272  }
273  if (debugging)
274  {
275  std::cout << "Max timevalue is " << *max_time << "\n";
276  std::cout << "There are " << max_time_vertices
277  << " vertices with max timeslice in the cell.\n";
278  std::cout << "Cell info is now " << c->info() << "\n";
279  std::cout << "---\n";
280  }

282  vertex_timevalues.clear();
283  }
