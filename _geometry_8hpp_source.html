240  [[nodiscard]] auto classify_cells(std::vector<Cell_handle> const& cells,
241  bool debugging = false) const
242  -> std::vector<Cell_handle>
243  {
244  Expects(cells.size() == number_of_cells_);
245  std::vector<Vertex_handle> cell_vertices;
246  cell_vertices.reserve(4);
247  std::vector<int> vertex_timevalues;
248  vertex_timevalues.reserve(4);
249  for (auto const& c : cells)
250  {
251  if (debugging) { std::cout << "Cell info was " << c->info() << '\n'; }
252 
253  for (int j = 0; j < 4; ++j)
254  {
255  cell_vertices.emplace_back(c->vertex(j));
256  vertex_timevalues.emplace_back(c->vertex(j)->info());
257  if (debugging)
258  {
259  std::cout << "Cell vertex " << j << " has timevalue "
260  << c->vertex(j)->info() << '\n';
261  }
262  }
263 
264  auto max_time =
265  std::max_element(vertex_timevalues.begin(), vertex_timevalues.end());
266  auto max_time_vertices =
267  std::count_if(cell_vertices.begin(), cell_vertices.end(),
268  [max_time](auto const& vertex) {
269  return vertex->info() == *max_time;
270  });
271 
272  switch (max_time_vertices)
273  {
274  case 1:
275  c->info() = static_cast<int>(Cell_type::THREE_ONE);
276  break;
277  case 2:
278  c->info() = static_cast<int>(Cell_type::TWO_TWO);
279  break;
280  case 3:

282  break;
283  default:
284  throw std::logic_error("Mis-classified cell.");
285  }
286  if (debugging)
287  {
288  std::cout << "Max timevalue is " << *max_time << "\n";
289  std::cout << "There are " << max_time_vertices

291  std::cout << "Cell info is now " << c->info() << "\n";
292  std::cout << "---\n";
293  }

296  }
