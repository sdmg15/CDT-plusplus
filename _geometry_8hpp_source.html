57  /// @brief Collect all finite cells of the triangulation
58  /// @tparam T Reference type of triangulation
59  /// @param universe Reference to triangulation
60  /// @return Vector of all the finite cells in the triangulation
61  template <typename T>
62  std::vector<Cell_handle> collect_cells(T& universe)
63  {
64  Expects(universe != nullptr);
65  std::vector<Cell_handle> init_cells;
66  init_cells.reserve(number_of_cells);
67  Delaunay3::Finite_cells_iterator cit;
68  for (cit = universe->finite_cells_begin();
69  cit != universe->finite_cells_end(); ++cit)
70  { init_cells.emplace_back(cit); }
71  Ensures(init_cells.size() == universe->number_of_finite_cells());
72  return init_cells;
73  }

75  /// @brief Filter simplices by Cell_type
76  /// @param cells_v The vector of simplices to filter
77  /// @param cell_t The Cell_type to filter by
78  /// @return A vector of Cell_type simplices
79  std::vector<Cell_handle> filter_cells(std::vector<Cell_handle> cells_v,
80  const Cell_type cell_t)
81  {
82  std::vector<Cell_handle> filtered_cells(cells_v.size());
83  auto it =
84  std::copy_if(cells_v.begin(), cells_v.end(), filtered_cells.begin(),
85  [&](auto const& cell) {
86  return cell->info() == static_cast<std::size_t>(cell_t);
87  });
88  filtered_cells.resize(std::distance(filtered_cells.begin(), it));
89  return filtered_cells;
90  }

102  Delaunay3::Finite_edges_iterator eit;

104  eit != universe->finite_edges_end(); ++eit)
105  {
106  Cell_handle ch = eit->first;
107  Edge_handle thisEdge{ch, ch->index(ch->vertex(eit->second)),
108  ch->index(ch->vertex(eit->third))};
109  init_edges.emplace_back(thisEdge);

111  Ensures(init_edges.size() == universe->number_of_finite_edges());
112  return init_edges;
