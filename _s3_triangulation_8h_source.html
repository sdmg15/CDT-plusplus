131  Delaunay::Finite_edges_iterator eit;
132  std::vector<Edge_handle> timelike_edges;
133  auto spacelike_edges = static_cast<std::uintmax_t>(0);
134 
135  // Iterate over all edges in the Delaunay triangulation
136  for (eit = universe_ptr->finite_edges_begin();
137  eit != universe_ptr->finite_edges_end(); ++eit) {
138  Cell_handle ch = eit->first;
139  // Get timevalues of vertices at the edge ends
140  auto time1 = ch->vertex(eit->second)->info();
141  auto time2 = ch->vertex(eit->third)->info();
142 
143  if (time1 != time2) { // We have a timelike edge
144  Edge_handle thisEdge{ch,
145  ch->index(ch->vertex(eit->second)),
146  ch->index(ch->vertex(eit->third))};
147  timelike_edges.emplace_back(thisEdge);

149  #ifdef DETAILED_DEBUGGING

152  << std::get<2>(thisEdge)
153  << std::endl;
154  #endif
155 
156  } else {
157  ++spacelike_edges;
158  } // endif
159  } // Finish iterating over edges

185 template<typename T>
186 auto classify_simplices(T &&universe_ptr) {
187  std::cout << "Classifying simplices...." << std::endl;
188  Delaunay::Finite_cells_iterator cit;
189  std::vector<Cell_handle> three_one;
190  std::vector<Cell_handle> two_two;
191  std::vector<Cell_handle> one_three;
192 
193  // Iterate over all cells in the Delaunay triangulation
194  for (cit = universe_ptr->finite_cells_begin();
195  cit != universe_ptr->finite_cells_end(); ++cit) {
196  std::list<std::uintmax_t> timevalues;
197  auto max_time = 0;
198  auto current_time = 0;
199  auto max_values = 0;
200  auto min_values = 0;
201  // Push every time value of every vertex into a list
202  for (auto i = 0; i < 4; ++i) {
203  timevalues.emplace_back(cit->vertex(i)->info());
204  }
205  // Now sort the list
206  timevalues.sort();
207  // The maximum timevalue is at the back of the list
208  max_time = timevalues.back();
209  timevalues.pop_back();
210  ++max_values;
211 
212  // Now pop the rest of the values
213  while (!timevalues.empty()) {
214  current_time = timevalues.back();
215  timevalues.pop_back();
216  (current_time == max_time) ? ++max_values : ++min_values;
217  }
218 
219  // Classify simplex using max_values and write to cit->info()
220  if (max_values == 3) {
221  cit->info() = 13;
222  one_three.emplace_back(cit);
223  } else if (max_values == 2) {
224  cit->info() = 22;
225  two_two.emplace_back(cit);
226  } else if (max_values == 1) {
227  cit->info() = 31;
228  three_one.emplace_back(cit);
229  } else {
230  throw
231  std::runtime_error("Invalid simplex in classify_simplices()!");
232  } // endif
233  } // Finish iterating over cells
234 
235  // Display results if debugging
236  #ifndef NDEBUG
237  std::cout << "There are " << three_one.size() << " (3,1) simplices and "
238  << two_two.size() << " (2,2) simplices" << std::endl;
239  std::cout << "and " << one_three.size() << " (1,3) simplices." << std::endl;
240  #endif
241 
242  return std::make_tuple(three_one, two_two, one_three);
243 } // classify_simplices()

290

293  cit != universe_ptr->finite_cells_end(); ++cit) {

299  #endif

324  }

326  #ifdef DETAILED_DEBUGGING

337  } else {

344  }
345  } // Finish iterating over cells
