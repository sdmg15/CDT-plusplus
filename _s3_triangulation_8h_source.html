120 auto fix_timeslices(T&& universe_ptr)
121 {
122  Delaunay::Finite_cells_iterator cit;
123  std::intmax_t min_time{0};
124  std::intmax_t max_time{0};
125  std::intmax_t valid{0};
126  std::intmax_t invalid{0};
127  std::intmax_t max_vertex{0};
128  std::set<Vertex_handle> deleted_vertices;
129 
130  // Iterate over all cells in the Delaunay triangulation
131  for (cit = universe_ptr->finite_cells_begin();
132  cit != universe_ptr->finite_cells_end(); ++cit)
133  {
134  if (cit->is_valid()) { // Valid cell
135  min_time = cit->vertex(0)->info();
136  max_time = min_time;
137 #ifdef DETAILED_DEBUGGING
138  bool this_cell_foliation_valid = true;
139 #endif
140  // Iterate over all vertices in the cell
141  for (auto i = 0; i < 4; ++i) {
142  auto current_time = cit->vertex(i)->info();
143 
144  // Classify extreme values
145  if (current_time < min_time) min_time = current_time;
146  if (current_time > max_time) {
147  max_time = current_time;
148  max_vertex = static_cast<intmax_t>(i);
149  }
150  } // Finish iterating over vertices
151  // There should be a difference of 1 between min_time and max_time
152  if (max_time - min_time != 1) {
153  invalid++;
154 #ifdef DETAILED_DEBUGGING
155  this_cell_foliation_valid = false;
156 #endif
157  // Single-threaded delete max vertex
158  // universe_ptr->remove(cit->vertex(max_vertex));
159 
160  // Parallel delete std::set of max_vertex for all invalid cells
161  deleted_vertices.emplace(cit->vertex(static_cast<int>(max_vertex)));
162  }
163  else
164  {
165  ++valid;
166  }
167 
168 #ifdef DETAILED_DEBUGGING
169  std::cout << "Foliation for cell is "
170  << ((this_cell_foliation_valid) ? "valid." : "invalid.\n");
171  for (auto i = 0; i < 4; ++i) {
172  std::cout << "Vertex " << i << " is " << cit->vertex(i)->point()
173  << " with timeslice " << cit->vertex(i)->info() << "\n";
174  }
175 #endif
176  }
177  else
178  {
179  throw std::runtime_error("Cell handle is invalid!");
180  // Or just remove the cell
181  // universe_ptr->tds().delete_cell(cit);
182  // This results in a possibly broken Delaunay triangulation
183  // Or possibly just delete a vertex in the cell,
184  // perhaps forcing a re-triangulation?
185  }
186  } // Finish iterating over cells
187 
188  // Delete invalid vertices
189  universe_ptr->remove(deleted_vertices.begin(), deleted_vertices.end());
190  // Check that the triangulation is still valid
191  // Turned off by -DCGAL_TRIANGULATION_NO_POSTCONDITIONS
192  // CGAL_triangulation_expensive_postcondition(universe_ptr->is_valid());
193  if (!universe_ptr->tds().is_valid())
194  throw std::logic_error("Delaunay tds invalid!");
195 
196 #ifndef NDEBUG
197  std::cout << "There are " << invalid << " invalid simplices and " << valid
198  << " valid simplices.\n";
199 #endif
200  return invalid == 0;
201 } // fix_timeslices

242 auto inline make_foliated_sphere(const std::intmax_t simplices,
243  const std::intmax_t timeslices)
244 {
245  // double radius{0};
246  const auto points_per_timeslice =
247  expected_points_per_simplex(DIMENSION, simplices, timeslices);
248  CGAL_triangulation_precondition(points_per_timeslice >= 2);
249  Causal_vertices causal_vertices;
250 
251  for (std::intmax_t i = 0; i < timeslices; ++i) {
252  auto radius = INITIAL_RADIUS + static_cast<double>(i) * RADIAL_FACTOR;
253  CGAL::Random_points_on_sphere_3<Point> gen{radius};
254  // At each radius, generate a sphere of random points proportional to area

258  causal_vertices.push_back(std::make_pair(*gen++, radius));
259  } // end j
