132 {
133 #ifndef NDEBUG
134  std::cout << "Classifying edges...." << std::endl;
135 #endif
136  Delaunay::Finite_edges_iterator eit;
137  std::vector<Edge_handle> timelike_edges;
138  std::vector<Edge_handle> spacelike_edges;
139 
140  // Iterate over all edges in the Delaunay triangulation
141  for (eit = universe_ptr->finite_edges_begin();
142  eit != universe_ptr->finite_edges_end(); ++eit)
143  {
144  Cell_handle ch = eit->first;
145  // Get timevalues of vertices at the edge ends
146  auto time1 = ch->vertex(eit->second)->info();
147  auto time2 = ch->vertex(eit->third)->info();
148 
149  // Make Edge_handle
150  Edge_handle thisEdge{
151  ch, static_cast<std::intmax_t>(ch->index(ch->vertex(eit->second))),
152  static_cast<std::intmax_t>(ch->index(ch->vertex(eit->third)))};
153 
154  if (time1 != time2)
155  { // We have a timelike edge
156  timelike_edges.emplace_back(thisEdge);
157 
158 #ifdef DETAILED_DEBUGGING
159  std::cout << "First vertex of edge is " << std::get<1>(thisEdge)
160  << " and second vertex of edge is " << std::get<2>(thisEdge)
161  << std::endl;
162 #endif
163  }
164  else
165  { // We have a spacelike edge
166  spacelike_edges.emplace_back(thisEdge);
167  } // endif
168  } // Finish iterating over edges
169 
170 // Display results if debugging
171 #ifndef NDEBUG
172  std::cout << "There are " << timelike_edges.size() << " timelike edges and "

204 
205  // Iterate over all cells in the Delaunay triangulation
206  for (cit = universe_ptr->finite_cells_begin();
207  cit != universe_ptr->finite_cells_end(); ++cit)
208  {
209  std::intmax_t max_values{0};
210  std::intmax_t min_values{0};
211  // Push every time value of every vertex into a list
212  std::intmax_t timevalues[4] = {
213  cit->vertex(0)->info(), cit->vertex(1)->info(), cit->vertex(2)->info(),
214  cit->vertex(3)->info(),
215  };
216  std::intmax_t max_time =
217  *std::max_element(std::begin(timevalues), std::end(timevalues));
218  for (auto elt : timevalues)

220  if (elt == max_time)
221  {
222  ++max_values;
223  }
224  else
225  {
226  ++min_values;
227  }
228  }
229 
230  // Classify simplex using max_values and write to cit->info()
231  if (min_values == 1 && max_values == 3)
232  {
233  cit->info() = 13;
234  one_three.emplace_back(cit);
235  }
236  else if (min_values == 2 && max_values == 2)

298 auto fix_timeslices(T&& universe_ptr)
299 { // NOLINT
300  Delaunay::Finite_cells_iterator cit;
301  std::intmax_t min_time{0};
302  std::intmax_t max_time{0};
303  std::intmax_t valid{0};
304  std::intmax_t invalid{0};
305  std::intmax_t max_vertex{0};
306  std::set<Vertex_handle> deleted_vertices;
307 
308  // Iterate over all cells in the Delaunay triangulation
309  for (cit = universe_ptr->finite_cells_begin();
310  cit != universe_ptr->finite_cells_end(); ++cit)
311  {
312  if (cit->is_valid())
313  { // Valid cell
314  min_time = cit->vertex(0)->info();
315  max_time = min_time;
316 #ifdef DETAILED_DEBUGGING
317  bool this_cell_foliation_valid = true;
318 #endif
319  // Iterate over all vertices in the cell
320  for (auto i = 0; i < 4; ++i)
321  {
322  auto current_time = cit->vertex(i)->info();
323 
324  // Classify extreme values
325  if (current_time < min_time) min_time = current_time;
326  if (current_time > max_time)
327  {
328  max_time = current_time;
329  max_vertex = static_cast<intmax_t>(i);
330  }
331  } // Finish iterating over vertices
332  // There should be a difference of 1 between min_time and max_time
333  if (max_time - min_time != 1)
334  {
335  invalid++;
336 #ifdef DETAILED_DEBUGGING
337  this_cell_foliation_valid = false;
338 #endif
339  // Single-threaded delete max vertex
340  // universe_ptr->remove(cit->vertex(max_vertex));
341 
342  // Parallel delete std::set of max_vertex for all invalid cells
343  deleted_vertices.emplace(cit->vertex(static_cast<int>(max_vertex)));
344  }
345  else
346  {
347  ++valid;
348  }
349 
350 #ifdef DETAILED_DEBUGGING
351  std::cout << "Foliation for cell is "
352  << ((this_cell_foliation_valid) ? "valid." : "invalid.")
353  << std::endl;
354  for (auto i = 0; i < 4; ++i)
355  {
356  std::cout << "Vertex " << i << " is " << cit->vertex(i)->point()
357  << " with timeslice " << cit->vertex(i)->info() << std::endl;
358  }
359 #endif
360  }
361  else
362  {
363  throw std::runtime_error("Cell handle is invalid!");
364  // Or just remove the cell
365  // universe_ptr->tds().delete_cell(cit);
366  // This results in a possibly broken Delaunay triangulation
367  // Or possibly just delete a vertex in the cell,
368  // perhaps forcing a re-triangulation?
369  }
370  } // Finish iterating over cells
371 
372  // Delete invalid vertices
373  universe_ptr->remove(deleted_vertices.begin(), deleted_vertices.end());
374  // Check that the triangulation is still valid
375  CGAL_triangulation_expensive_postcondition(universe_ptr->is_valid());
376 
377 #ifndef NDEBUG
378  std::cout << "There are " << invalid << " invalid simplices and " << valid
379  << " valid simplices." << std::endl;
380 #endif
381  return invalid == 0;
382 } // fix_timeslices
