57  /// Parse user options from the given option string.
58  ///
59  /// @param doc The usage string
60  /// @param argv The user-supplied arguments
61  /// @param help Whether to end early if '-h' or '--help' is in the argv
62  /// @param version Whether to end early if '--version' is in the argv
63  /// @param options_first Whether options must precede all args (true), or if args and options
64  /// can be arbitrarily mixed.
65  ///
66  /// @throws DocoptLanguageError if the doc usage string had errors itself
67  /// @throws DocoptExitHelp if 'help' is true and the user has passed the '--help' argument
68  /// @throws DocoptExitVersion if 'version' is true and the user has passed the '--version' argument
69  /// @throws DocoptArgumentError if the user's argv did not match the usage patterns
70  std::map<std::string, value> DOCOPT_API docopt_parse(std::string const& doc,
