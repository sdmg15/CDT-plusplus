335  static inline std::vector<PatternList> transform(PatternList pattern)
336  {
337  std::vector<PatternList> result;
338 
339  std::vector<PatternList> groups;
340  groups.emplace_back(std::move(pattern));
341 
342  while(!groups.empty()) {
343  // pop off the first element
344  auto children = std::move(groups[0]);
345  groups.erase(groups.begin());
346 
347  // find the first branch node in the list
348  auto child_iter = std::find_if(children.begin(), children.end(), [](std::shared_ptr<Pattern> const& p) {
349  return dynamic_cast<BranchPattern const*>(p.get());
350  });
351 
352  // no branch nodes left : expansion is complete for this grouping
353  if (child_iter == children.end()) {
354  result.emplace_back(std::move(children));
355  continue;
356  }
357 
358  // pop the child from the list
359  auto child = std::move(*child_iter);
360  children.erase(child_iter);
361 
362  // expand the branch in the appropriate way
363  if (auto* either = dynamic_cast<Either*>(child.get())) {
364  // "[e] + children" for each child 'e' in Either
365  for(auto const& eitherChild : either->children()) {
366  PatternList group = { eitherChild };
367  group.insert(group.end(), children.begin(), children.end());
368 
369  groups.emplace_back(std::move(group));
370  }
371  } else if (auto* oneOrMore = dynamic_cast<OneOrMore*>(child.get())) {
372  // child.children * 2 + children
373  auto const& subchildren = oneOrMore->children();
374  PatternList group = subchildren;
375  group.insert(group.end(), subchildren.begin(), subchildren.end());
376  group.insert(group.end(), children.begin(), children.end());
377 
378  groups.emplace_back(std::move(group));
379  } else { // Required, Optional, OptionsShortcut
380  auto* branch = dynamic_cast<BranchPattern*>(child.get());
381 
382  // child.children + children
383  PatternList group = branch->children();
384  group.insert(group.end(), children.begin(), children.end());
385 
386  groups.emplace_back(std::move(group));
387  }
388  }
389 
390  return result;
391  }

393  inline void BranchPattern::fix_repeating_arguments()
394  {
395  std::vector<PatternList> either = transform(children());
396  for(auto const& group : either) {
397  // use multiset to help identify duplicate entries
398  std::unordered_multiset<std::shared_ptr<Pattern>, PatternHasher> group_set {group.begin(), group.end()};
399  for(auto const& e : group_set) {
400  if (group_set.count(e) == 1)
401  continue;
402 
403  auto* leaf = dynamic_cast<LeafPattern*>(e.get());
404  if (!leaf) continue;
405 
406  bool ensureList = false;
407  bool ensureInt = false;
408 
409  if (dynamic_cast<Command*>(leaf)) {
410  ensureInt = true;
411  } else if (dynamic_cast<Argument*>(leaf)) {
412  ensureList = true;
413  } else if (auto* o = dynamic_cast<Option*>(leaf)) {
414  if (o->argCount()) {
415  ensureList = true;
416  } else {
417  ensureInt = true;
418  }
419  }
420 
421  if (ensureList) {
422  std::vector<std::string> newValue;
423  if (leaf->getValue().isString()) {
424  newValue = split(leaf->getValue().asString());
425  }
426  if (!leaf->getValue().isStringList()) {
427  leaf->setValue(value{newValue});
428  }
429  } else if (ensureInt) {
430  leaf->setValue(value{0});
431  }
432  }
433  }

436  inline bool LeafPattern::match(PatternList& left, std::vector<std::shared_ptr<LeafPattern>>& collected) const
437  {
438  auto match = single_match(left);
439  if (!match.second) {
440  return false;
441  }
442 
443  left.erase(left.begin()+static_cast<std::ptrdiff_t>(match.first));
444 
445  auto same_name = std::find_if(collected.begin(), collected.end(), [&](std::shared_ptr<LeafPattern> const& p) {
446  return p->name()==name();
447  });
448  if (getValue().isLong()) {
449  long val = 1;
450  if (same_name == collected.end()) {
451  collected.push_back(match.second);
452  match.second->setValue(value{val});
453  } else if ((**same_name).getValue().isLong()) {
454  val += (**same_name).getValue().asLong();
455  (**same_name).setValue(value{val});
456  } else {
457  (**same_name).setValue(value{val});
458  }
459  } else if (getValue().isStringList()) {
460  std::vector<std::string> val;
461  if (match.second->getValue().isString()) {
462  val.push_back(match.second->getValue().asString());
463  } else if (match.second->getValue().isStringList()) {
464  val = match.second->getValue().asStringList();
465  } else {
467  }
468 
469  if (same_name == collected.end()) {
470  collected.push_back(match.second);
471  match.second->setValue(value{val});
472  } else if ((**same_name).getValue().isStringList()) {
473  std::vector<std::string> const& list = (**same_name).getValue().asStringList();
474  val.insert(val.begin(), list.begin(), list.end());
475  (**same_name).setValue(value{val});
476  } else {
477  (**same_name).setValue(value{val});
478  }
479  } else {
480  collected.push_back(match.second);
481  }
482  return true;

521  inline Option Option::parse(std::string const& option_description)
522  {
523  std::string shortOption, longOption;
524  int argcount = 0;
525  value val { false };
526 
527  auto double_space = option_description.find(" ");
528  auto options_end = option_description.end();
529  if (double_space != std::string::npos) {
530  options_end = option_description.begin() + static_cast<std::ptrdiff_t>(double_space);
531  }
532 
533  static const std::regex pattern {"(-{1,2})?(.*?)([,= ]|$)"};
534  for(std::sregex_iterator i {option_description.begin(), options_end, pattern, std::regex_constants::match_not_null},
535  e{};
536  i != e;
537  ++i)
538  {
539  std::smatch const& match = *i;
540  if (match[1].matched) { // [1] is optional.
541  if (match[1].length()==1) {
542  shortOption = "-" + match[2].str();
543  } else {
544  longOption = "--" + match[2].str();
545  }
546  } else if (match[2].length() > 0) { // [2] always matches.
547  std::string m = match[2];
548  argcount = 1;
549  } else {
550  // delimeter
551  }
552 
553  if (match[3].length() == 0) { // [3] always matches.
554  // Hit end of string. For some reason 'match_not_null' will let us match empty
555  // at the end, and then we'll spin in an infinite loop. So, if we hit an empty
556  // match, we know we must be at the end.
557  break;
558  }
559  }
560 
561  if (argcount) {
562  std::smatch match;
563  if (std::regex_search(options_end, option_description.end(),
564  match,
565  std::regex{R"(\[default: (.*)\])", std::regex::icase}))
566  {
567  val = match[1].str();
568  }
569  }

623  ++times;

626  firstLoop = false;

629  }

644  {

653  bool matched = pattern->match(l, c);
654  if (matched) {
