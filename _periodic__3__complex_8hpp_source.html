30 void make_random_T3_simplicial_complex(PDT* T3,
31  int number_of_simplices) noexcept
32 {
33  typedef CGAL::Creator_uniform_3<double, PDT::Point> Creator;
34  CGAL::Random random(7);
35  CGAL::Random_points_in_cube_3<PDT::Point, Creator> in_cube(.5, random);
36 
37  /// We can't directly pick number of simplices as we can in S3
38  /// but heuristically a point has <6 simplices
39  int n = number_of_simplices / 6;
40  std::vector<PDT::Point> pts;
41 
42  // Generate random points
43  for (int i = 0; i < n; i++)
44  {
45  PDT::Point p = *in_cube;
46  in_cube++;
47  pts.emplace_back(PDT::Point(p.x() + .5, p.y() + .5, p.z() + .5));
48  }
49 
50  // Iterator range insertion using spatial sorting and dummy point heuristic
51  T3->insert(pts.begin(), pts.end(), true);
52 
53  assert(T3->dimension() == 3);
54  assert(T3->is_valid());
55 }
