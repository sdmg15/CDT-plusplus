31 /// @brief Try a (2,3) move
32 ///
33 /// This function performs the (2,3) move by converting the facet
34 /// between a (3,1) simplex and a (2,2) simplex into its dual edge.
35 ///
36 /// @tparam T The manifold type
37 /// @param universe A SimplicialManifold
38 /// @param to_be_moved The **Cell_handle** that is tried
39 /// @return A boolean value whether the move succeeded
40 template <typename T>
41 auto try_23_move(T&& universe, Cell_handle to_be_moved)
42 {
43  auto flipped = false;
44  // Try every facet of the cell
45  for (auto i = 0; i < 4; ++i)
46  {
47  if (universe.triangulation->flip(to_be_moved, i))
48  {
49 #ifndef NDEBUG
50  std::cout << "Facet " << i << " was flippable.\n";
51 #endif
52 
53  flipped = true;
54  break;
55  }
56  else
57  {
58 #ifndef NDEBUG
59  std::cout << "Facet " << i << " was not flippable.\n";
60 #endif
61  }
62  }
63  return flipped;
64 } // try_23_move()

67 ///
68 /// A (2,3) moves adds a (2,2) simplex and a timelike edge.
69 ///
70 /// This function calls **try_23_move()** until it succeeds; the
71 /// triangulation is no longer Delaunay.
72 ///
73 /// @tparam T1 The manifold type
74 /// @tparam T2 The type of the tuple holding attempted moves
75 /// @param universe A SimplicialManifold
76 /// @param attempted_moves A tuple holding a count of the attempted moves
77 /// @return The SimplicialManifold after the move has been made

82  std::cout << "Attempting (2,3) move.\n";
83 #endif
84 
85  auto movable_two_two_cells{universe.geometry->two_two};
86 
87  auto not_flipped = true;
88  while (not_flipped)
89  {
90  if (movable_two_two_cells.size() == 0)
91  { throw std::domain_error("No (2,3) move is possible."); }
92  // Pick out a random (2,2) which ranges from 0 to size()-1
93  auto choice = generate_random_signed(0, movable_two_two_cells.size() - 1);
94 
95  Cell_handle to_be_moved = universe.geometry->two_two[choice];
96  if (try_23_move(universe, to_be_moved)) not_flipped = false;
97 
98  // Remove trial cell
99  movable_two_two_cells.erase(movable_two_two_cells.begin() + choice);
100 
101  // Increment the (2,3) move counter
102  ++attempted_moves[0];
103  }
104  // Uses return value optimization and allows chaining function calls
105  // return std::move(universe);
106  return std::forward<T1>(universe);

185 /// neighbor of a (1,3) cell. That is, the i-th neighboring cell must be
186 /// a (3,1) cell, and of course the base cell must be a (1,3). This preserves

188 /// general case.

231 ///
232 /// A (2,6) move adds 2 (1,3) simplices and 2 (3,1) simplices for a
233 /// total of 3 (1,3) simplices and 3 (3,1) simplices.
234 /// It also adds 2 timelike edges and 3 spacelike edges, for a total
235 /// of 8 timelike edges and 6 spacelike edges.
236 ///
237 /// This function performs the (2,6) move by picking a random (1,3) simplex
238 /// from GeometryInfo. The **find_26_movable()** function finds the
239 /// index of the neighboring (3,1) simplex, **neighboring_31_index**, and
240 /// **has_neighbor()** is used to check the results.
241 ///
242 /// After some other values are gathered for debugging purposes,
243 /// the **v_center** vertex is inserted into the facet delineated by
244 /// **neighboring_31_index** using **tds().insert_in_facet()**.
245 /// Finally, the centroid of the common face calculated using
246 /// **CGAL::centroid()** and assigned to **v_center**, along with a
247 /// timevalue taken from one of the vertices of the common face.
248 /// @todo Check Euler condition on insert, ensure not inserting outside
249 ///
250 /// @image html 26.png
251 /// @image latex 26.eps width=7cm
252 ///
253 /// @tparam T1 The manifold type
254 /// @tparam T2 The type of the tuple holding attempted moves
255 /// @param universe A SimplicialManifold
256 /// @param attempted_moves A tuple holding a count of the attempted moves
257 /// of each type given by the **move_type** enum
258 /// @return The SimplicialManifold{} after the move has been made
