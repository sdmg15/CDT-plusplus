189  [[nodiscard]] auto fix_timeslices() -> bool
190  {
191  int min_time{0};
192  int max_time{0};
193  int valid{0};
194  int invalid{0};
195  int max_vertex{0};
196  std::set<Vertex_handle> deleted_vertices;
197  // Iterate over all cells in the Delaunay triangulation
198  for (Delaunay3::Finite_cells_iterator cit = delaunay_.finite_cells_begin();
199  cit != delaunay_.finite_cells_end(); ++cit)
200  {
201  if (cit->is_valid())
202  { // Valid cell
203  min_time = cit->vertex(0)->info();
204  max_time = min_time;
205 #ifdef DETAILED_DEBUGGING
206  bool this_cell_foliation_valid = true;
207 #endif
208  // Iterate over all vertices in the cell
209  for (int i = 0; i < 4; ++i)
210  {
211  auto current_time = cit->vertex(i)->info();
212 
213  // Classify extreme values
214  if (current_time < min_time) min_time = current_time;
215  if (current_time > max_time)
216  {
217  max_time = current_time;
218  max_vertex = i;
219  }
220  } // Finish iterating over vertices
221  // There should be a difference of 1 between min_time and max_time
222  if (max_time - min_time != 1)
223  {
224  invalid++;
225 #ifdef DETAILED_DEBUGGING
226  this_cell_foliation_valid = false;
227 #endif
228  // Single-threaded delete max vertex
229  // universe_ptr->remove(cit->vertex(max_vertex));
230 
231  // Parallel delete std::set of max_vertex for all invalid cells
232  deleted_vertices.emplace(cit->vertex(max_vertex));
233  }
234  else
235  {
236  ++valid;
237  }
238 
239 #ifdef DETAILED_DEBUGGING
240  std::clog << "Foliation for cell is "
241  << ((this_cell_foliation_valid) ? "valid." : "invalid.\n");
242  for (auto i = 0; i < 4; ++i)
243  {
244  std::clog << "Vertex " << i << " is " << cit->vertex(i)->point()
245  << " with timeslice " << cit->vertex(i)->info() << "\n";
246  }
247 #endif
248  }
249  else
250  {
251  throw std::runtime_error("Cell handle is invalid!");
252  }
253  } // Finish iterating over cells
254 
255  // Delete invalid vertices
256  delaunay_.remove(deleted_vertices.begin(), deleted_vertices.end());
257  // Check that the triangulation is still valid
258  // Turned off by -DCGAL_TRIANGULATION_NO_POSTCONDITIONS
259  // CGAL_triangulation_expensive_postcondition(universe_ptr->is_valid());
260  // if (!_delaunay.tds().is_valid())
261  // throw std::logic_error("Delaunay tds invalid!");
262  Ensures(delaunay_.tds().is_valid());
263  Ensures(delaunay_.is_valid());
264 
265 #ifndef NDEBUG
266  std::cout << "There are " << invalid << " invalid simplices and " << valid
267  << " valid simplices.\n";
268 #endif
269  return invalid == 0;
270  } // fix_timeslices
